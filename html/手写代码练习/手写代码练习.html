package cn.dc.prac;

import org.junit.jupiter.api.Test;

import java.util.Arrays;

public class 手写代码练习 {
    /*
    冒泡排序：
        内层循环执行1次，比较相邻2个数并排序。完整循环一次会把极值找出并放到极端位置
        外层循环要比较length - 1次
        优化1：如果1次内层循环比较一次没有发生交换，说明当前集合已经有序，直接退出即可
        优化2：记录前一次内层循环最后发生交换的位置，因为后面的位置已经有序了。下次内层循环只需要排序前面部分即可。
     */
    @Test
    public void testBubbleSort() {
        Integer[] nums = {3, 4, 7, 5, 0, 6, 6, 12, -1};
        bubbleSort(nums);
        System.out.println(Arrays.toString(nums));
        Arrays.stream(nums).forEach(num -> System.out.println(num));
    }

    //正序
    private void bubbleSort(Integer[] nums) {
        boolean changed = false;
        int k = nums.length - 1;
        int pos = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            for (int j = 0; j < k; j++) {
                if (nums[j] > nums[j + 1]) {
                    nums[j] = nums[j] ^ nums[j + 1];
                    nums[j + 1] = nums[j] ^ nums[j + 1];
                    nums[j] = nums[j] ^ nums[j + 1];
                    changed = true;
                    pos = j;
                }
            }
            if (!changed) return;
            k = pos;
        }
    }

    /*
    快速排序
        设置1个基准点，然后通过左右游标互换的方式来实现批量交换来分成左右2个子序列，然后递归每个子序列。是bubbleSort的升级版
     */
    @Test
    public void testQS() {
        Integer[] nums = {3, 4, 7, 5, 0, 6, 6, 12, -1};
        quickSort(nums, 0, nums.length - 1);
        System.out.println(Arrays.toString(nums));
        Arrays.stream(nums).forEach(num -> System.out.println(num));
    }

    private void quickSort(Integer[] nums, int left, int right) {
        if (left < right) {
            int index = partitionAndGetPivotIndex(nums, left, right);
            quickSort(nums, left, index - 1);
            quickSort(nums, index + 1, right);
        }
    }

    private int partitionAndGetPivotIndex(Integer[] nums, int left, int right) {
        int pivot = nums[left];
        while (left < right) {
            while (left < right && nums[right] >= pivot) right--;
            nums[left] = nums[right];
            while (left < right && nums[left] <= pivot) left++;
            nums[right] = nums[left];
        }
        nums[left] = pivot;
        return left;
    }
    /*
    二分查找
        跟中间值比较，通过递归不断缩小比较范围
     */
    @Test
    public void testBinarySearch(){
        int[] nums = {2,3,5,7,12,23,33,34,56};
        System.out.println(binarySearch(nums,56));
    }
    private int binarySearch(int[] nums,int target){
        int start = 0, end = nums.length - 1, middle = 0;
        while (start <= end) {
            middle = start + (end - start) / 2;
            if (target == nums[middle]) return middle;
            else if (target > nums[middle]) start = middle + 1;
            else end = middle - 1;
        }
        return -1;
    }
}

二、              复制文件夹

class CopyFolder1 {
    /*
    复制文件夹，要用递归，不断地执行：遍历子文件，是文件就起名并复制，是文件夹就调自己的逻辑
    最外层负责调度，内层就是被调的原子逻辑
    使用recursion
        创建复制一个文件夹和里面文件的方法，然后判断里面的文件是否是文件夹，如果是文件夹，递归调用自己

    注意：先把根目录创建了，因为createNewFile这个方法不会自动创建自己的父目录，跟mkdirs不一样。
     */
    void copyFolder(String inFilepath, String outFilePath) {
        /*
        1.准备工作
            创建路径的File对象
            判断outFile是否存在，如不存在，创建
         */
        File inFile = new File(inFilepath);
        File outFile = new File(outFilePath);
        if (!outFile.exists()) outFile.mkdirs();
        /*
        2.遍历inFile中的文件，如果是文件，就调用copySingleFile复制，如果是文件夹，调自己
         */
        File[] files = inFile.listFiles();
        for (File file : files) {
            if (file.isDirectory()) {
                copyFolder(file.getPath(), outFilePath + "\\" + file.getName());
            } else {
                copyFile(file.getPath(), outFilePath + "\\" + file.getName());

            }
        }
    }

    /*
    复制单个文件
     */
    private void copyFile(String inFilepath, String outFilePath)  {
        try (InputStream is = new FileInputStream(inFilepath);
             OutputStream os = new FileOutputStream(outFilePath);) {
            //创建缓冲区
            byte[] bytes = new byte[1024];
            //读到缓冲区，如果读完返回-1
            int flag = -1;
            while ((flag = is.read(bytes)) != -1) {
                os.write(bytes, 0, flag);
                os.flush();
            }
        } catch (FileNotFoundException e) {
            e.printStackTrace();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
    @Test
    public void testCopyFile(){
        copyFolder("C:\\Users\\Administrator\\Desktop\\MobileFile","C:\\Users\\Administrator\\Desktop\\MobileFile1");
    }
}

package cn.dc.string;

import org.junit.jupiter.api.Test;

public class 反转字符串 {
    /*
    将字符串切分为字符，倒着遍历拼接成新字符串
     */
    @Test
    public void ttt1() {
        String message = "我们黄种人";
        StringBuilder sbuilder = new StringBuilder();
        for (int i = message.length() - 1; i >= 0; i--)
            sbuilder.append(message.charAt(i));

        System.out.println(sbuilder);
    }
}
